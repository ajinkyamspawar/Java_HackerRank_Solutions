#16: Java Substring Comparison

                         #to split given string in the form of substring 
for(int i = 1;i<=length-k;i++){
	first = s.substring(i,i+k);
	
	if(first.compareTo(largest)>0)
	largest = first;

	if(first.compareTo(smallest)<0)
	smallest = first;
